// js/auth.js const SUPABASE_URL = 'https://yourproject.supabase.co'; const SUPABASE_KEY = 'your-anon-key'; async function register(activationKey, username, password) { // 1. Check if activation key is valid and unused const keyCheck = await fetch(`${SUPABASE_URL}/rest/v1/activation_keys?key_value=eq.${activationKey}&is_used=eq.false`, { headers: { 'apikey': SUPABASE_KEY } }); const keys = await keyCheck.json(); if (keys.length === 0) { alert('Invalid or already used activation key!'); return; } // 2. Hash password (never store plain text) const hashBuffer = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(password)); const hash = Array.from(new Uint8Array(hashBuffer)).map(b => b.toString(16).padStart(2,'0')).join(''); // 3. Create user const userRes = await fetch(`${SUPABASE_URL}/rest/v1/users`, { method: 'POST', headers: { 'apikey': SUPABASE_KEY, 'Content-Type': 'application/json', 'Prefer': 'return=representation' }, body: JSON.stringify({ username, password_hash: hash }) }); const user = await userRes.json(); // 4. Mark activation key as used await fetch(`${SUPABASE_URL}/rest/v1/activation_keys?key_value=eq.${activationKey}`, { method: 'PATCH', headers: { 'apikey': SUPABASE_KEY, 'Content-Type': 'application/json' }, body: JSON.stringify({ is_used: true, used_by: user[0].id, used_at: new Date().toISOString() }) }); alert('Registered! You can now login.'); } async function login(username, password) { const hashBuffer = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(password)); const hash = Array.from(new Uint8Array(hashBuffer)).map(b => b.toString(16).padStart(2,'0')).join(''); const res = await fetch(`${SUPABASE_URL}/rest/v1/users?username=eq.${username}&password_hash=eq.${hash}&is_banned=eq.false`, { headers: { 'apikey': SUPABASE_KEY } }); const users = await res.json(); if (users.length > 0) { // Store session sessionStorage.setItem('sf_user', JSON.stringify(users[0])); // Show download page showDownloads(); } else { alert('Invalid credentials or account banned!'); } }
