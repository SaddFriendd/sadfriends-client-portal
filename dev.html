// js/dev.js const DEV_PASSWORD = 'your-secret-dev-password-here'; // only you know this const SUPABASE_SERVICE_KEY = 'your-service-role-key'; // full access function devLogin(pass) { if (pass !== DEV_PASSWORD) { alert('Access denied'); return; } sessionStorage.setItem('dev_auth', '1'); loadDevPanel(); } function generateKey() { const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789'; let key = ''; for (let i = 0; i < 24; i++) { if (i > 0 && i % 4 === 0) key += '-'; key += chars[Math.floor(Math.random() * chars.length)]; } return key; } async function saveAndDisplayKey() { const key = generateKey(); // Save to Supabase await fetch(`${SUPABASE_URL}/rest/v1/activation_keys`, { method: 'POST', headers: { 'apikey': SUPABASE_SERVICE_KEY, 'Content-Type': 'application/json' }, body: JSON.stringify({ key_value: key }) }); document.getElementById('current-key').textContent = key; } // When you click "Copy", it saves to clipboard then generates a new one async function copyKey() { const key = document.getElementById('current-key').textContent; await navigator.clipboard.writeText(key); await saveAndDisplayKey(); // auto-generate next key } async function loadUsers() { const res = await fetch(`${SUPABASE_URL}/rest/v1/users?order=created_at.desc`, { headers: { 'apikey': SUPABASE_SERVICE_KEY } }); return await res.json(); } async function banUser(userId) { await fetch(`${SUPABASE_URL}/rest/v1/users?id=eq.${userId}`, { method: 'PATCH', headers: { 'apikey': SUPABASE_SERVICE_KEY, 'Content-Type': 'application/json' }, body: JSON.stringify({ is_banned: true }) }); }
